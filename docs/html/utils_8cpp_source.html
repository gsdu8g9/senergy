29 std::string Utils::EncodeHostname(const std::string &hostname)
30 {
31  if(hostname.empty())
32  return "";
33 
34  /*
35  www.google.come -> 3www6google3com
36  where 3, 6, 3 are NOT ASCII but just decimals
37  */
38 
39  size_t hostname_len = hostname.size();
40 
41  std::string new_hostname = " ";
42  new_hostname += hostname;
43 
44  int previous_dot_pos = 0;
45  int segment_count = 0;
46 
47  for(size_t i = 1; i < hostname_len + 1; ++i)
48  {
49  if(i == hostname_len) // last char
50  new_hostname[previous_dot_pos] = ++segment_count;
51 
52  if(new_hostname[i] == '.')
53  {
54  new_hostname[previous_dot_pos] = segment_count;
55  previous_dot_pos = (int)i;
56  segment_count = 0;
57  continue;
58  }
59 
60  new_hostname[i] = tolower(new_hostname[i]);
61  segment_count++;
62  }
63 
64  return new_hostname;
65 }

67 std::string Utils::DecodeHostname(const std::string &hostname)
68 {
69  if(hostname.empty())
70  return "";
71 
72  /*
73  3www6google3com -> www.google.com
74  where 3, 6, 3 are NOT ASCII, but just decimals
75  */
76 
77  int hostname_len = (int) hostname.size();
78  std::string new_hostname = "";
79 
80  for(int i = 0; i < hostname_len; ++i)
81  {
82  int segment_length = hostname[i];
83 
84  for(int y = 0; y < segment_length; ++y)
85  {
86  new_hostname += hostname[i + 1];
87  i++;
88  }
89 
90  if(i != (hostname_len - 1)) // skip last
91  new_hostname += ".";
92  }
93 
94  return new_hostname;
95 }
