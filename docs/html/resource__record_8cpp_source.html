38 bool ResourceRecord::Deserialize(ByteBuffer &buffer)
39 {
40  if(buffer.GetRemainingSize() < 4)
41  return false;
42 
43  m_hostname = buffer.ReadString(256);
44 
45  if(m_hostname.size() < 2)
46  return false;
47 
48  // we do not support pointers..
49  if(Utils::IsDomainPointer(m_hostname))
50  return false;
51 
52  m_hostname = Utils::DecodeHostname(m_hostname);
53 
54  if(buffer.GetRemainingSize() < 4)
55  return false;
56 
57  m_type = (ResourceRecordType) buffer.ReadUnsignedShort();
58  m_class = (ResourceRecordClass) buffer.ReadUnsignedShort();
59  m_ttl = buffer.ReadUnsignedInt();
60  m_rd_length = buffer.ReadUnsignedShort();
61 
62  m_type = (ResourceRecordType) Utils::NetworkToHostByteOrder((unsigned short)m_type);
63  m_class = (ResourceRecordClass) Utils::NetworkToHostByteOrder((unsigned short)m_class);
64  m_ttl = Utils::NetworkToHostByteOrder(m_ttl);
65  m_rd_length = Utils::NetworkToHostByteOrder(m_rd_length);
66  return true;
67 }

69 bool ResourceRecord::Serialize(ByteBuffer &buffer)
70 {
71  if(m_hostname.empty() || m_rd_length <= 0)
72  return false;
73 
74  std::string encoded_hostname = Utils::EncodeHostname(m_hostname);
75  buffer.Write(encoded_hostname);
76 
77  buffer.Write(Utils::HostToNetworkByteOrder((unsigned short)m_type));
78  buffer.Write(Utils::HostToNetworkByteOrder((unsigned short)m_class));
79  buffer.Write(Utils::HostToNetworkByteOrder(m_ttl));
80  buffer.Write(Utils::HostToNetworkByteOrder(m_rd_length));
81  return true;
82 }
