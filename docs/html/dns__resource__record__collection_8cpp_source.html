41 bool ResourceRecordCollection::Deserialize(MessageHeader &header, ByteBuffer &buffer)
42 {
43  int amount_answers = (int) header.Fields.AnswerCount;
44  int amount_nameservers = (int) header.Fields.NameServerCount;
45  int amount_additional = (int) header.Fields.AdditionalCount;
46  int amount_total = amount_answers + amount_nameservers + amount_additional;
47 
48  for(int i = 0; i < amount_total; ++i)
49  {
50  if(buffer.GetRemainingSize() < ResourceRecordBase::GetMinimalSize())
51  return false;
52 
53  ResourceRecordBasePtr base_record = ResourceRecordBase::Create();
54  if(!base_record->Deserialize(buffer))
55  return false;
56 
57  if(buffer.GetRemainingSize() < base_record->GetResourceSize())
58  return false;
59 
60  ResourceRecordInterfacePtr record = m_mapper.ApplyMapping(base_record->GetType(), base_record);
61  if(!record)
62  {
63  buffer.IncreasePosition(base_record->GetResourceSize());
64  continue;
65  }
66 
67  if(!record->Deserialize(buffer))
68  return false;
69 
70  m_records[record->GetType()].Add(record);
71  }
72 
73  return true;
74 }
