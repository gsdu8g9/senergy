64 bool MessageQuestion::Deserialize(ByteBuffer &buffer)
65 {
66  if(buffer.GetRemainingSize() < 3)
67  return false;
68 
69  m_hostname = Utils::ReadHostnameFromBuffer(buffer);
70 
71  if(buffer.GetRemainingSize() < 2)
72  return false;
73 
74  m_type = buffer.ReadUnsignedShort();
75  m_class = buffer.ReadUnsignedShort();
76 
77  m_type = Utils::NetworkToHostByteOrder(m_type);
78  m_class = Utils::NetworkToHostByteOrder(m_class);
79  return true;
80 }

82 bool MessageQuestion::Serialize(ByteBuffer &buffer)
83 {
84  if(m_hostname.empty())

86 
87  std::string encoded_hostname = Utils::EncodeHostname(m_hostname);
88 
89  buffer.Write(encoded_hostname);

91  buffer.Write(Utils::HostToNetworkByteOrder(m_class));
92
