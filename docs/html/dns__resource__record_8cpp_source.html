43 bool ResourceRecord::Deserialize(ByteBuffer &buffer)
44 {
45  if(buffer.GetRemainingSize() < 4)
46  return false;
47 
48  m_hostname = Utils::ReadHostnameFromBuffer(buffer);
49 
50  if(m_hostname.size() <= 0)
51  return false;
52 
53  if(buffer.GetRemainingSize() < 4)
54  return false;
55 
56  m_type = (ResourceRecordType) buffer.ReadUnsignedShort();
57  m_class = (ResourceRecordClass) buffer.ReadUnsignedShort();
58  m_ttl = buffer.ReadUnsignedInt();
59  m_rd_length = buffer.ReadUnsignedShort();
60 
61  m_type = (ResourceRecordType) Utils::NetworkToHostByteOrder((unsigned short)m_type);
62  m_class = (ResourceRecordClass) Utils::NetworkToHostByteOrder((unsigned short)m_class);
63  m_ttl = Utils::NetworkToHostByteOrder(m_ttl);
64  m_rd_length = Utils::NetworkToHostByteOrder(m_rd_length);
65 
66  return true;
67 }

69 bool ResourceRecord::Serialize(ByteBuffer &buffer)
70 {
71  if(m_hostname.empty() || m_rd_length <= 0)
72  return false;
73 
74  std::string encoded_hostname = Utils::EncodeHostname(m_hostname);
75  buffer.Write(encoded_hostname);
76 
77  buffer.Write(Utils::HostToNetworkByteOrder((unsigned short)m_type));
78  buffer.Write(Utils::HostToNetworkByteOrder((unsigned short)m_class));
79  buffer.Write(Utils::HostToNetworkByteOrder(m_ttl));
80  buffer.Write(Utils::HostToNetworkByteOrder(m_rd_length));
81  return true;
82 }
