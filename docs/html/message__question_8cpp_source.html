100 std::string MessageQuestion::__encode_hostname(const std::string &hostname)
101 {
102  if(hostname.empty())
103  return "";
104 
105  /*
106  www.google.come -> 3www6google3com
107  where 3, 6, 3 are NOT ASCII but just decimals
108  */
109 
110  size_t hostname_len = hostname.size();
111 
112  std::string new_hostname = " ";
113  new_hostname += hostname;
114 
115  int previous_dot_pos = 0;
116  int segment_count = 0;
117 
118  for(size_t i = 1; i < hostname_len + 1; ++i)
119  {
120  if(i == hostname_len) // last char
121  new_hostname[previous_dot_pos] = ++segment_count;
122 
123  if(new_hostname[i] == '.')
124  {
125  new_hostname[previous_dot_pos] = segment_count;
126  previous_dot_pos = (int)i;
127  segment_count = 0;
128  continue;
129  }
130 
131  new_hostname[i] = tolower(new_hostname[i]);
132  segment_count++;
133  }
134 
135  return new_hostname;
136 }

138 std::string MessageQuestion::__decode_hostname(const std::string &hostname)
139 {
140  if(hostname.empty())
141  return "";
142 
143  /*
144  3www6google3com -> www.google.com
145  where 3, 6, 3 are NOT ASCII, but just decimals
146  */
147 
148  int hostname_len = (int) hostname.size();
149  std::string new_hostname = "";
150 
151  for(int i = 0; i < hostname_len; ++i)
152  {
153  int segment_length = hostname[i];
154 
155  for(int y = 0; y < segment_length; ++y)
156  {
157  new_hostname += hostname[i + 1];
158  i++;
159  }
160 
161  if(i != (hostname_len - 1)) // skip last
162  new_hostname += ".";
163  }
164 
165  return new_hostname;
166 }
